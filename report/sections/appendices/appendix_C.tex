\section{Random forest}
\begin{verbatim}
        "name": "Random Forest",
        "filename": "random_forest",
        "model": GridSearchCV(
            RandomForestClassifier(random_state=1),
            [
                {
                    "n_estimators": [150, 295, 350],
                    "max_features": ["auto"],
                    "min_samples_split": [8, 12],
                    "max_depth": [15, 18, 21],
                }
            ],
            cv=5,
        )
\end{verbatim}

\section{Support vector machine}
\begin{verbatim}
        "name": "Support Vector Machine",
